Vue指令 vs 原生JavaScript

Vue.js - v-bind
1. 属性绑定
<!-- Vue模板 --> <img v-bind:src="imageUrl" alt="图片" /> <div :class="cssClass">动态样式</div> <button :disabled="isDisabled">按钮</button> <!-- 数据 --> data() { return { imageUrl: '/path/to/image.jpg', cssClass: 'active', isDisabled: true } }
原生JavaScript
setAttribute / 属性赋值
// 获取元素 const img = document.querySelector('img'); const div = document.querySelector('div'); const button = document.querySelector('button'); // 设置属性 img.setAttribute('src', imageUrl); // 或者: img.src = imageUrl; div.className = cssClass; // 或者: div.setAttribute('class', cssClass); button.disabled = isDisabled; // 或者: button.setAttribute('disabled', isDisabled);
Vue.js - v-on
2. 事件监听
<!-- Vue模板 --> <button v-on:click="handleClick">点击</button> <button @click="handleClick">简写</button> <input @keyup.enter="onEnter" /> <form @submit.prevent="onSubmit"></form> <!-- 方法 --> methods: { handleClick(event) { console.log('按钮被点击', event); }, onEnter() { /* ... */ }, onSubmit() { /* ... */ } }
原生JavaScript
addEventListener
// 获取元素 const button = document.querySelector('button'); const input = document.querySelector('input'); const form = document.querySelector('form'); // 添加事件监听 button.addEventListener('click', handleClick); input.addEventListener('keyup', function(e) { if (e.key === 'Enter') { onEnter(); } }); form.addEventListener('submit', function(e) { e.preventDefault(); // 阻止默认行为 onSubmit(); }); function handleClick(event) { console.log('按钮被点击', event); }
Vue.js - v-if/v-show
3. 条件渲染
<!-- Vue模板 --> <div v-if="isVisible"> 条件为真时显示 </div> <div v-else-if="isOther"> 其他条件 </div> <div v-else> 默认显示 </div> <p v-show="isShown">切换显示/隐藏</p> <!-- 数据 --> data() { return { isVisible: true, isOther: false, isShown: true } }
原生JavaScript
DOM显示/隐藏操作
const div1 = document.getElementById('conditional'); const div2 = document.getElementById('other'); const div3 = document.getElementById('default'); const p = document.getElementById('toggle'); // v-if 等价操作（添加/移除DOM） if (isVisible) { document.body.appendChild(div1); } else { div1.remove(); } // v-show 等价操作（显示/隐藏） if (isShown) { p.style.display = 'block'; } else { p.style.display = 'none'; } // 或者使用类切换 p.classList.toggle('hidden', !isShown);
Vue.js - v-for
4. 列表渲染
<!-- Vue模板 --> <ul> <li v-for="item in items" :key="item.id"> {{ item.name }} - {{ item.price }} </li> </ul> <!-- 带索引 --> <div v-for="(item, index) in items" :key="index"> {{ index }}: {{ item.name }} </div> <!-- 数据 --> data() { return { items: [ { id: 1, name: '苹果', price: 5 }, { id: 2, name: '香蕉', price: 3 }, { id: 3, name: '橙子', price: 4 } ] } }
原生JavaScript
循环创建DOM元素
const items = [ { id: 1, name: '苹果', price: 5 }, { id: 2, name: '香蕉', price: 3 }, { id: 3, name: '橙子', price: 4 } ]; const ul = document.createElement('ul'); // 使用forEach循环 items.forEach(item => { const li = document.createElement('li'); li.textContent = `${item.name} - ${item.price}`; li.setAttribute('data-id', item.id); // 相当于key ul.appendChild(li); }); // 或使用for循环 for (let i = 0; i < items.length; i++) { const div = document.createElement('div'); div.textContent = `${i}: ${items[i].name}`; document.body.appendChild(div); } document.body.appendChild(ul);
Vue.js - v-model
5. 双向数据绑定
<!-- Vue模板 --> <input v-model="message" /> <textarea v-model="content"></textarea> <select v-model="selected"> <option value="a">选项A</option> <option value="b">选项B</option> </select> <p>输入内容: {{ message }}</p> <!-- 数据 --> data() { return { message: '初始值', content: '', selected: 'a' } }
原生JavaScript
事件监听 + 属性绑定
let message = '初始值'; let content = ''; let selected = 'a'; const input = document.querySelector('input'); const textarea = document.querySelector('textarea'); const select = document.querySelector('select'); const p = document.querySelector('p'); // 设置初始值 input.value = message; textarea.value = content; select.value = selected; // 监听输入事件，更新数据 input.addEventListener('input', (e) => { message = e.target.value; p.textContent = `输入内容: ${message}`; }); textarea.addEventListener('input', (e) => { content = e.target.value; }); select.addEventListener('change', (e) => { selected = e.target.value; });
