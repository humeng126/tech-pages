🎯 Vue模板语法详解

📝 {{ }} - 插值语法（文本插值）

💡 作用

{{ }} 用于在HTML模板中显示JavaScript表达式的结果，类似于模板字符串的 ${}

Vue模板写法
<template>
  <div>
    <h1>{{ message }}</h1>
    <p>用户名: {{ user.name }}</p>
    <p>年龄: {{ user.age }} 岁</p>
    <p>总价: {{ price * quantity }} 元</p>
    <p>问候: {{ greeting() }}</p>
    <p>时间: {{ new Date().toLocaleString() }}</p>
  </div>
</template>

<script>
export default {
  data() {
    return {
      message: '欢迎使用Vue!',
      user: { name: '张三', age: 25 },
      price: 99,
      quantity: 2
    }
  },
  methods: {
    greeting() {
      return `你好, ${this.user.name}!`;
    }
  }
}
</script>
原生JavaScript等价
// 数据
const data = {
  message: '欢迎使用Vue!',
  user: { name: '张三', age: 25 },
  price: 99,
  quantity: 2
};

function greeting() {
  return `你好, ${data.user.name}!`;
}

// 手动更新DOM内容
function updateDOM() {
  document.querySelector('h1').textContent = data.message;
  
  document.querySelector('.username').textContent = 
    `用户名: ${data.user.name}`;
  
  document.querySelector('.age').textContent = 
    `年龄: ${data.user.age} 岁`;
  
  document.querySelector('.total').textContent = 
    `总价: ${data.price * data.quantity} 元`;
  
  document.querySelector('.greeting').textContent = 
    `问候: ${greeting()}`;
  
  document.querySelector('.time').textContent = 
    `时间: ${new Date().toLocaleString()}`;
}

// 数据变化时需要手动调用
updateDOM();
渲染结果

  <p>用户名: 张三</p>
  <p>年龄: 25 岁</p>

{{ }} 支持的内容 示例 说明
变量 {{ message }} 显示data中的变量
对象属性 {{ user.name }} 访问对象的属性
数组元素 {{ items[0] }} 访问数组的元素
表达式计算 {{ price * quantity }} 进行数学运算
方法调用 {{ formatDate() }} 调用methods中的方法
三元运算符 {{ age >= 18 ? '成年' : '未成年' }} 条件判断
JavaScript内置对象 {{ Math.round(price) }} 使用JS原生方法
🔗 : - 属性绑定（v-bind简写）

💡 作用

: 是 v-bind: 的简写，用于动态绑定HTML属性的值

Vue属性绑定
<template>
  <div>
    <!-- 图片源绑定 -->
    <img :src="imageUrl" :alt="imageAlt" />
    
    <!-- CSS类绑定 -->
    <div :class="cssClasses">动态样式</div>
    <div :class="{ active: isActive, disabled: isDisabled }">
      条件样式
    </div>
    
    <!-- 内联样式绑定 -->
    <p :style="{ color: textColor, fontSize: textSize + 'px' }">
      动态样式
    </p>
    
    <!-- 表单属性绑定 -->
    <input :value="inputValue" :disabled="isFormDisabled" />
    
    <!-- 链接绑定 -->
    <a :href="linkUrl" :target="linkTarget">点击链接</a>
    
    <!-- 自定义属性 -->
    <div :data-id="userId" :title="tooltipText">
      用户信息
    </div>
  </div>
</template>

<script>
export default {
  data() {
    return {
      imageUrl: '/avatar.jpg',
      imageAlt: '用户头像',
      cssClasses: 'container active',
      isActive: true,
      isDisabled: false,
      textColor: '#ff6b6b',
      textSize: 18,
      inputValue: '初始值',
      isFormDisabled: false,
      linkUrl: 'https://example.com',
      linkTarget: '_blank',
      userId: 123,
      tooltipText: '这是提示信息'
    }
  }
}
</script>
原生JavaScript等价
// 数据
const data = {
  imageUrl: '/avatar.jpg',
  imageAlt: '用户头像',
  cssClasses: 'container active',
  isActive: true,
  isDisabled: false,
  textColor: '#ff6b6b',
  textSize: 18,
  inputValue: '初始值',
  isFormDisabled: false,
  linkUrl: 'https://example.com',
  linkTarget: '_blank',
  userId: 123,
  tooltipText: '这是提示信息'
};

// 手动设置属性
function bindAttributes() {
  const img = document.querySelector('img');
  img.src = data.imageUrl;
  img.alt = data.imageAlt;
  
  const dynamicDiv = document.querySelector('.dynamic');
  dynamicDiv.className = data.cssClasses;
  
  const conditionalDiv = document.querySelector('.conditional');
  conditionalDiv.className = '';
  if (data.isActive) conditionalDiv.classList.add('active');
  if (data.isDisabled) conditionalDiv.classList.add('disabled');
  
  const styledP = document.querySelector('p');
  styledP.style.color = data.textColor;
  styledP.style.fontSize = data.textSize + 'px';
  
  const input = document.querySelector('input');
  input.value = data.inputValue;
  input.disabled = data.isFormDisabled;
  
  const link = document.querySelector('a');
  link.href = data.linkUrl;
  link.target = data.linkTarget;
  
  const userDiv = document.querySelector('.user-info');
  userDiv.setAttribute('data-id', data.userId);
  userDiv.title = data.tooltipText;
}

// 数据变化时需要手动调用
bindAttributes();

绑定类型 Vue写法 完整写法 JavaScript等价
属性绑定 :src="url" v-bind:src="url" element.src = url
类绑定 :class="className" v-bind:class="className" element.className = className
样式绑定 :style="styleObj" v-bind:style="styleObj" Object.assign(element.style, styleObj)
布尔属性 :disabled="isDisabled" v-bind:disabled="isDisabled" element.disabled = isDisabled
自定义属性 :data-id="id" v-bind:data-id="id" element.setAttribute('data-id', id)

⚠️ 重要注意事项

❌ 常见错误

  • {{ }} 不能用在HTML属性中
  • :onclick 是错误的，应该用 @click
  • {{ }} 中不能写复杂的语句
  • 不要在 {{ }} 中修改数据

✅ 正确用法

  • {{ message }} - 显示文本
  • :src="imageUrl" - 绑定属性
  • {{ user.name.toUpperCase() }} - 简单表达式
  • 使用computed计算复杂逻辑

🚀 总结对比

{{ }} 插值语法

用于显示数据内容，相当于JavaScript中的模板字符串或textContent赋值

: 属性绑定

用于动态设置HTML属性，相当于JavaScript中的setAttribute或属性赋值
